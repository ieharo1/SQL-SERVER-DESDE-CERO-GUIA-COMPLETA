Introducción a las Tablas

Una tabla es la estructura fundamental para almacenar datos en una base de datos relacional. Consiste en filas (registros) y columnas (campos).

💡 Nota: Cada tabla debe tener un nombre único dentro de la base de datos.

Tipos de Datos Comunes

Tipo de Dato Descripción Ejemplo
INT Número entero 1, 100, -50
DECIMAL(10,2) Número decimal con precisión 99.99, 1234.56
VARCHAR(50) Texto de longitud variable 'Hola', 'SQL Server'
NVARCHAR(100) Texto Unicode de longitud variable 'México', 'José'
DATE Fecha (YYYY-MM-DD) 2025-01-15
DATETIME Fecha y hora 2025-01-15 14:30:00
BIT Valor booleano (0/1) 0, 1, TRUE, FALSE

Crear una Tabla

SQL 
-- Crear tabla de empleados
CREATE TABLE Empleados (
    EmpleadoID INT PRIMARY KEY IDENTITY(1,1),
    Nombre NVARCHAR(100) NOT NULL,
    Apellido NVARCHAR(100) NOT NULL,
    Email NVARCHAR(255) UNIQUE,
    Telefono NVARCHAR(20),
    FechaNacimiento DATE,
    Salario DECIMAL(10,2),
    Activo BIT DEFAULT 1,
    FechaContratacion DATETIME DEFAULT GETDATE()
);
GO

Restricciones (Constraints)

SQL 
-- Tabla con múltiples restricciones
CREATE TABLE Productos (
    ProductoID INT CONSTRAINT PK_Productos PRIMARY KEY,
    SKU NVARCHAR(50) CONSTRAINT UQ_Productos_SKU UNIQUE NOT NULL,
    Nombre NVARCHAR(200) NOT NULL,
    Precio DECIMAL(10,2) CONSTRAINT CK_Productos_Precio CHECK (Precio >= 0),
    Stock INT CONSTRAINT DF_Productos_Stock DEFAULT 0,
    CategoriaID INT CONSTRAINT FK_Productos_Categorias 
        FOREIGN KEY REFERENCES Categorias(CategoriaID)
);
GO

Tipos de Restricciones:

  • PRIMARY KEY - Clave primaria (única y no nula)
  • FOREIGN KEY - Clave foránea (referencia a otra tabla)
  • UNIQUE - Valores únicos en la columna
  • CHECK - Valida que los datos cumplan una condición
  • DEFAULT - Valor predeterminado
  • NOT NULL - La columna no puede ser nula

Columna Identity (Autoincremental)

SQL 
-- Identity(semilla, incremento)
CREATE TABLE Pedidos (
    PedidoID INT IDENTITY(1,1) PRIMARY KEY,  -- Comienza en 1, incrementa de 1 en 1
    FechaPedido DATETIME DEFAULT GETDATE(),
    Total DECIMAL(10,2)
);

-- Identity con diferente incremento
CREATE TABLE Facturas (
    FacturaID INT IDENTITY(1000,1) PRIMARY KEY,  -- Comienza en 1000
    Monto DECIMAL(10,2)
);
GO

Modificar una Tabla (ALTER TABLE)

SQL 
-- Agregar una columna
ALTER TABLE Empleados
ADD Direccion NVARCHAR(255);
GO

-- Modificar tipo de dato de columna
ALTER TABLE Empleados
ALTER COLUMN Telefono NVARCHAR(30);
GO

-- Eliminar una columna
ALTER TABLE Empleados
DROP COLUMN Direccion;
GO

-- Agregar restricción
ALTER TABLE Empleados
ADD CONSTRAINT CK_Empleados_Salario CHECK (Salario > 0);
GO

-- Eliminar restricción
ALTER TABLE Empleados
DROP CONSTRAINT CK_Empleados_Salario;
GO

Eliminar una Tabla

⚠️ Advertencia: DROP TABLE elimina permanentemente la tabla y todos sus datos.
SQL 
-- Eliminar tabla
DROP TABLE Empleados;
GO

-- Eliminar solo si existe
DROP TABLE IF EXISTS Empleados;
GO

Ver Información de Tablas

SQL 
-- Listar todas las tablas
SELECT TABLE_NAME 
FROM INFORMATION_SCHEMA.TABLES 
WHERE TABLE_TYPE = 'BASE TABLE'
ORDER BY TABLE_NAME;
GO

-- Ver columnas de una tabla
SELECT 
    COLUMN_NAME,
    DATA_TYPE,
    CHARACTER_MAXIMUM_LENGTH,
    IS_NULLABLE,
    COLUMN_DEFAULT
FROM INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_NAME = 'Empleados'
ORDER BY ORDINAL_POSITION;
GO

-- Ver restricciones
SELECT 
    tc.CONSTRAINT_NAME,
    tc.CONSTRAINT_TYPE,
    kcu.COLUMN_NAME
FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS tc
JOIN INFORMATION_SCHEMA.KEY_COLUMN_USAGE kcu 
    ON tc.CONSTRAINT_NAME = kcu.CONSTRAINT_NAME
WHERE tc.TABLE_NAME = 'Empleados';
GO

Ejercicio Práctico

Crea las tablas para un sistema de biblioteca:

SQL - Ejercicio 
-- 1. Tabla Libros (LibroID, ISBN, Titulo, Autor, AnioPublicacion, Precio, Stock)
-- 2. Tabla Socios (SocioID, Nombre, Email, FechaRegistro, Activo)
-- 3. Tabla Prestamos (PrestamoID, LibroID, SocioID, FechaPrestamo, FechaDevolucion)

-- Solución:
CREATE TABLE Libros (
    LibroID INT PRIMARY KEY IDENTITY(1,1),
    ISBN NVARCHAR(20) UNIQUE,
    Titulo NVARCHAR(200) NOT NULL,
    Autor NVARCHAR(100),
    AnioPublicacion INT,
    Precio DECIMAL(10,2) CHECK (Precio >= 0),
    Stock INT DEFAULT 0
);
GO

CREATE TABLE Socios (
    SocioID INT PRIMARY KEY IDENTITY(1,1),
    Nombre NVARCHAR(150) NOT NULL,
    Email NVARCHAR(255) UNIQUE,
    FechaRegistro DATETIME DEFAULT GETDATE(),
    Activo BIT DEFAULT 1
);
GO

CREATE TABLE Prestamos (
    PrestamoID INT PRIMARY KEY IDENTITY(1,1),
    LibroID INT FOREIGN KEY REFERENCES Libros(LibroID),
    SocioID INT FOREIGN KEY REFERENCES Socios(SocioID),
    FechaPrestamo DATETIME DEFAULT GETDATE(),
    FechaDevolucion DATETIME
);
GO
